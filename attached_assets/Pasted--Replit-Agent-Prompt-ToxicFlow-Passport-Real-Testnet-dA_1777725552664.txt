# Replit Agent Prompt: ToxicFlow Passport — Real Testnet dApp

## Project Overview

Build **ToxicFlow Passport** — a trust-aware swap and liquidity protection protocol deployed on Sepolia testnet. This is a complete, production-quality dApp, not a prototype or mock. Every integration must be real.

**Core concept:** Most liquidity pools charge all wallets the same swap fee. ToxicFlow gives each wallet a behavior-based passport. When a wallet swaps through a ToxicFlow-managed pool, the hook reads the passport tier and applies the appropriate fee class automatically.

Fee classes:
- **Trusted**: 10 bps (0.10%)
- **Neutral**: 30 bps (0.30%)
- **Toxic**: 80 bps (0.80%)

ToxicFlow is not only a dApp — it is a passport and tier resolver layer that external AMMs, DEXes, and swap interfaces can integrate.

---

## Absolute Rules

1. **No mock data in the core app.** Fake transactions, fake wallet addresses, fake storage success, fake KeeperHub execution, and fake swap results are forbidden in all pages except the explicitly guided demo flow.
2. **No private keys in client-side code.** All sensitive operations use server-side API routes.
3. **Never label a fake token as real USDC.** If real Sepolia USDC cannot be used, label the fallback "Test Token" — never "Mock USDC" or "USDC."
4. **Never say ToxicFlow changes gas fees.** ToxicFlow controls the pool swap fee only. Network gas fees are separate and shown separately in the UI.
5. **No "Injected Wallet" as a user-facing label.** Detect the actual wallet provider and show its name.

### `contracts/src/PassportRegistry.sol`

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.26;

import {IPassportRegistry} from "./interfaces/IPassportRegistry.sol";

contract PassportRegistry is IPassportRegistry {
    uint8 public constant TIER_NEUTRAL = 0;
    uint8 public constant TIER_TRUSTED = 1;
    uint8 public constant TIER_TOXIC   = 2;

    address public admin;
    mapping(address => uint8)  private _tiers;
    mapping(address => bool)   private _registered;

    event PassportRegistered(address indexed trader);
    event TierUpdated(address indexed trader, uint8 tier, address indexed setBy);
    event AdminTransferred(address indexed prev, address indexed next);

    error NotAdmin();
    error InvalidTier();
    error AlreadyRegistered();
    error LengthMismatch();

    modifier onlyAdmin() { if (msg.sender != admin) revert NotAdmin(); _; }

    constructor(address _admin) { admin = _admin; emit AdminTransferred(address(0), _admin); }

    // Any wallet can self-register — receives neutral tier by default
    function selfRegister() external {
        if (_registered[msg.sender]) revert AlreadyRegistered();
        _registered[msg.sender] = true;
        _tiers[msg.sender] = TIER_NEUTRAL;
        emit PassportRegistered(msg.sender);
        emit TierUpdated(msg.sender, TIER_NEUTRAL, msg.sender);
    }

    function setTier(address trader, uint8 tier) external onlyAdmin {
        if (tier > TIER_TOXIC) revert InvalidTier();
        _tiers[trader] = tier;
        _registered[trader] = true;
        emit TierUpdated(trader, tier, msg.sender);
    }

    function batchSetTiers(address[] calldata traders, uint8[] calldata tiers) external onlyAdmin {
        if (traders.length != tiers.length) revert LengthMismatch();
        for (uint256 i = 0; i < traders.length; i++) {
            if (tiers[i] > TIER_TOXIC) revert InvalidTier();
            _tiers[traders[i]] = tiers[i];
            _registered[traders[i]] = true;
            emit TierUpdated(traders[i], tiers[i], msg.sender);
        }
    }

    function getTier(address trader) external view override returns (uint8) { return _tiers[trader]; }
    function hasPassport(address trader) external view override returns (bool) { return _registered[trader]; }

    // Tier resolver interface
    function getTraderTier(address trader) external view returns (uint8) { return _tiers[trader]; }
    function hasActivePassport(address trader) external view returns (bool) { return _registered[trader]; }

    function transferAdmin(address newAdmin) external onlyAdmin {
        emit AdminTransferred(admin, newAdmin);
        admin = newAdmin;
    }
}
```

### `contracts/src/ToxicFlowHook.sol`

Read installed `lib/v4-core/src/interfaces/IHooks.sol`, `lib/v4-core/src/libraries/Hooks.sol`, and `lib/v4-periphery/src/base/hooks/BaseHook.sol` before implementing. Use the exact installed signatures — do not guess.

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.26;

// Import BaseHook — check installed path in lib/v4-periphery/src/
import {BaseHook} from "@uniswap/v4-periphery/src/base/hooks/BaseHook.sol";
import {Hooks} from "@uniswap/v4-core/src/libraries/Hooks.sol";
import {IPoolManager} from "@uniswap/v4-core/src/interfaces/IPoolManager.sol";
import {PoolKey} from "@uniswap/v4-core/src/types/PoolKey.sol";
import {PoolId, PoolIdLibrary} from "@uniswap/v4-core/src/types/PoolId.sol";
import {BeforeSwapDelta, BeforeSwapDeltaLibrary} from "@uniswap/v4-core/src/types/BeforeSwapDelta.sol";
import {IPassportRegistry} from "./interfaces/IPassportRegistry.sol";

contract ToxicFlowHook is BaseHook {
    using PoolIdLibrary for PoolKey;

    // Uniswap v4 pip units: 1,000,000 = 100%
    uint24 public constant FEE_TRUSTED = 1_000;  // 10 bps
    uint24 public constant FEE_NEUTRAL = 3_000;  // 30 bps
    uint24 public constant FEE_TOXIC   = 8_000;  // 80 bps

    IPassportRegistry public immutable registry;

    // Phase 9C: KeeperHub subscribes to this event
    // Phase 9D: Indexer reads this for fee record creation
    event SwapFeeApplied(
        bytes32 indexed poolId,
        address indexed trader,
        uint8  tier,
        uint24 appliedFee,
        int128 amountSpecified,
        bool   hadPassport,
        uint256 blockNumber
    );

    constructor(IPoolManager _manager, address _registry) BaseHook(_manager) {
        registry = IPassportRegistry(_registry);
    }

    function getHookPermissions() public pure override returns (Hooks.Permissions memory) {
        return Hooks.Permissions({
            beforeInitialize: false, afterInitialize: false,
            beforeAddLiquidity: false, afterAddLiquidity: false,
            beforeRemoveLiquidity: false, afterRemoveLiquidity: false,
            beforeSwap: true, afterSwap: false,
            beforeDonate: false, afterDonate: false,
            beforeSwapReturnDelta: false, afterSwapReturnDelta: false,
            afterAddLiquidityReturnDelta: false, afterRemoveLiquidityReturnDelta: false
        });
    }

    function beforeSwap(
        address,
        PoolKey calldata key,
        IPoolManager.SwapParams calldata params,
        bytes calldata
    ) external override onlyPoolManager returns (bytes4, BeforeSwapDelta, uint24) {
        address trader = tx.origin;
        uint8 tier = registry.getTier(trader);
        bool passport = registry.hasPassport(trader);
        uint24 fee = _fee(tier);

        emit SwapFeeApplied(
            PoolId.unwrap(key.toId()),
            trader, tier, fee,
            params.amountSpecified,
            passport,
            block.number
        );

        return (BaseHook.beforeSwap.selector, BeforeSwapDeltaLibrary.ZERO_DELTA, fee);
    }

    function _fee(uint8 tier) internal pure returns (uint24) {
        if (tier == 1) return FEE_TRUSTED;
        if (tier == 2) return FEE_TOXIC;
        return FEE_NEUTRAL;
    }

    // Resolver interface — any wallet can call
    function previewFee(address trader) external view returns (uint24 fee, uint8 tier, bool hasPassport) {
        tier = registry.getTier(trader);
        hasPassport = registry.hasPassport(trader);
        fee = _fee(tier);
    }

    function getTraderTier(address trader) external view returns (uint8) { return registry.getTier(trader); }
    function hasActivePassport(address trader) external view returns (bool) { return registry.hasPassport(trader); }
}
```

```ts
  if (name.includes('zerion')) return { displayName: 'Zerion', iconKey: 'zerion' }
  if (name.includes('coinbase')) return { displayName: 'Coinbase Wallet', iconKey: 'coinbase' }
  if (id === 'injected' || name === 'injected') {
    if (typeof window !== 'undefined') {
      const eth = (window as any).ethereum
      if (eth?.isRabby) return { displayName: 'Rabby', iconKey: 'rabby' }
      if (eth?.isOKExWallet || eth?.isOKX) return { displayName: 'OKX Wallet', iconKey: 'okx' }
      if (eth?.isMetaMask) return { displayName: 'MetaMask', iconKey: 'metamask' }
    }
    return { displayName: 'Browser Wallet', iconKey: 'browser' }
  }
  return { displayName: connector.name, iconKey: 'browser' }
}
```
